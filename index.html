      // HTML
<table>
    <tr>

        <th>Column Header 2</th>
    </tr>
    <tr>
        <td>Row 2 Column 1</td>
        <td>Row 2 Column 2</td>
    </tr>
</table>
                        
// CSS

// Puts a 2 pixel solid white border around the whole table
table {
    border: 2px solid #fff;
}

// Gives the Header Cells a background color of #fc4d36, 
    and increases the font to 150% the size
th {
    background-color: rgb(252, 77, 54);
    font-size: 150%;
}

// Adds a 15px padding around the contents 
    of the heading and regular cells of the table.
th,
td {
    padding: 15px;
    text-align: center;
}

// Makes the even numbered regular cells have 
    a color of #00ffff51 (the 51 is alpha transparency)
tr:nth-child(even) {
    background-color: #00ffff51;
}

// Makes the Header cells (1st cells) text yellow color 
tr:nth-child(1) {
    color: yellow;
}
